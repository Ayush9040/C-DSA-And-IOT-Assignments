1. Write a recursive function to print first N natural numbers

Ans. #include<stdio.h>

    void printN(int);

    int main()
    {
        printN(20);
        return 0;
    }

    void printN(int n)
    {
        if(n>0)
        {
            printN(n-1);
            printf("%d ",n);
        }
    }

8. Write a recursive function to print binary of a given decimal number

Ans. #include<stdio.h>

    void binary(int);

    int main()
    {
        binary(43);
        return 0;
    }

    void binary(int n)
    {
        if(n==1)
        {
            printf("1");
        }
        else
        {
            binary(n/2);
            printf("%d",n%2);
        }
    }
